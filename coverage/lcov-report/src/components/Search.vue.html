<template>
    <div class="search mx-auto">
        <p class="search__label my-0 pl-5">
            {{ $t('Check Your city weather') }}
        </p>
        <form
            id="searchForm"
            @submit.prevent="citySearch"
            class="search__form d-flex"
        >
            <input
                id="searchField"
                v-model="city"
                class="search__field px-5"
                type="text"
                :placeholder="$t('city name')"
            />
            <button
                :disable="!city"
                class="search__submit text-uppercase"
                type="submit"
            >
                {{ $t('search') }}
            </button>
        </form>
        <p
            v-if="getError"
            class="search__error pl-5 caption"
        >
            {{ getError }}
        </p>
    </div>
</template>
 
<script>
import { mapGetters } from 'vuex';
 
export default {
  name: 'Search',
  data() {
    return {
      city: '',
    };
  },
  computed: {
    ...mapGetters(['getError']),
  },
  methods: {
    citySearch() {
      this.$store.dispatch('citySearch', this.city);
 
      this.city = '';
    },
  },
};
</script>
